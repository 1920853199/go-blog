下载安装

https://github.com/1920853199/go-blog.git

创建数据库，导入数据库文件

修改配置

                                    
                                        #conf/app.conf

                                        appname = go-blog

                                        httpport = 8088

                                        runmode = dev

                                        EnableAdmin = false

                                        sessionon = true

                                        url = 127.0.0.1:8088

                                        view = default



                                        limit = 10

                                        title = Go Blog

                                        autograph = 如今的我，谈不上幸福，也谈不上不幸。



                                        [db]

                                        dbUser = root

                                        dbPass = root

                                        dbHost = 127.0.0.1

                                        dbPort = 3306

                                        dbName = blog



                                        [redis]

                                        rHost = 127.0.0.1

                                        rPort = 6379



                                        [wechat]

                                        AppID = xxxxxxx

                                        AppSecret = xxxxxxx

                                        Token = xxxxxxx

                                        EncodingAESKey = xxxxxxx

编译运行

go build -o hello hello.go
./hello
